Various Methods in console function

In javascript, the console is an object which provides access to the browser debugging console. We can open a console in web browser by using: Ctrl + Shift + K for windows and Command + Option + K for Mac. The console object provides us with several different methods, like :
console.log()
console.error()
console.warn()
console.clear()
console.time() and timeEnd()
console.table()
console.count()
console.group() and groupEnd()
console.custom console.logs

console.log()
Mainly used to log(print) the output to the console.
We can put any type inside the log(), be it a string, array, object, boolean etc.
/* 1. console.log() :- Outputs a message to the console. */
console.log("1. This operation is successful."); // This will print "1. This operation is successful " in the console.
console.info()
/* console.info() :- Outputs an informational message to the console. */
console.info("This is info!"); //will display "this is info" as an information.
console.warn()
/* console.warn():- Outputs a warning message to the console.*/
console.warn("Lookout!"); //will display "Lookout!" as a warning.
console.error()
/* console.error() :- Outputs an error meesage to the console. */
console.error("Something went wrong.") //display message in danger color.
console.trace()
/*console.trace() :- Outputs a stack trace to the console.*/
function hello(name = "Ghrushma"){ console.trace("Name: ", name); return `Hello ${name}`;}
console.assert()
/*console.assert() :- Writes an error message to the console if the assertion is false. */
console.assert(2==2,'This is correct'); //display no message
console.assert(1==2,'This is wrong'); //Will display message
console.count()
/* console.count() :- Logs the number of times that this particular call to count has been called.*/
console.count("Count Number"); //Count Number: 1 console.count("Count Number"); //Count Number: 2
console.count("Count Number"); //Count Number: 3
console.count("Count Number"); //Count Number: 4
console.group()
/*console.group() :- Creates a new inline group in the console.
This indents following console message by an additional level, until console.groupEnd() is called.
console.groupEnd() :- Exits the current inline group in the console. */
console.group("Names");
console.log("Ghrushma");
console.log("Priya");
console.group("Roll Number");
console.log("48");
console.log("39");
console.groupEnd();
console.log("Kavita")
; console.groupEnd();
console.table()
/*console.table() :- Display tabular data as a table */
var information={ Name:"Ghrushma", Age:23, Email:"xyz@gmail.com"
console.table(information);
console.time()
/* console.time() :- Starts a timer (can track how long an operation takes.
console.timeEnd() :- Stops the times that was previously started by console.time() */
